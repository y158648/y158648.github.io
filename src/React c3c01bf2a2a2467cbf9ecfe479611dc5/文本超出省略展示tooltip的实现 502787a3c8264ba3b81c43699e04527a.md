# 文本超出省略展示tooltip的实现

- 带 `react` 逻辑的实现

```tsx
import { Tooltip } from "...";
import classNames from "classnames";
import React, {
  HTMLAttributes,
  useRef,
  useState,
  PropsWithChildren,
} from "react";

/** 单行文本超出展示tooltip */
const EllipsisText = ({
  children,
  className,
}: PropsWithChildren<HTMLAttributes<HTMLDivElement>>) => {
  const [showEllipsis, setShowEllipsis] = useState(false);
  const wrapperRef = useRef<HTMLDivElement>(null);

  const content = (
    <div
      className={classNames(
        "relative w-full overflow-hidden text-ellipsis whitespace-nowrap",
        className
      )}
      ref={wrapperRef}
    >
      {children}
      <div
        className="invisible absolute whitespace-nowrap"
        ref={(dom) => {
          if (dom && wrapperRef.current) {
            setShowEllipsis(
              dom.clientHeight > wrapperRef.current.clientHeight ||
                dom.clientWidth > wrapperRef.current.clientWidth
            );
          }
        }}
      >
        {children}
      </div>
    </div>
  );

  return showEllipsis ? <Tooltip title={children}>{content}</Tooltip> : content;
};

export default EllipsisText;
```
